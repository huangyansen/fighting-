#include <stdio.h>
#include <algorithm>
#include <iostream>
#include <string.h>
#include <string>
using namespace std;

void merge_(int s,int m,int e,int a[]){//将数组a[s...m]与a[m+1...e]重排序后放入a数组
    int b[e-s+2];//b中元素下标由0到e-s+1
    int i=s,j=m+1,k=0;
    while(i<=m&&j<=e){
        if(a[i]<a[j]){
            b[k++]=a[i++];
        }
        else{
            b[k++]=a[j++];
        }
    }

    while(i<=m)
        b[k++]=a[i++];
    while(j<=e)
        b[k++]=a[j++];

    for(i=s,j=0;i<=e;++i,++j)
        a[i]=b[j];
}

void Merge_sort(int s,int e,int a[]){
    if(s<e){
        int mid=(s+e)/2;
        Merge_sort(s,mid,a);
        Merge_sort(mid+1,e,a);
        merge_(s,mid,e,a);
    }
}


int main(){
    int n;
    int a[105];
    cout<<"input the number of array:\n";
    cin>>n;
    for(int i=1;i<=n;++i)
        cin>>a[i];

    Merge_sort(1,n,a);
    cout<<"After sort:\n";
    for(int i=1;i<=n;++i)
        cout<<a[i]<<endl;

    return 0;
}
